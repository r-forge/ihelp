흔히 지도상에서 도시와 도시사이를 연결하는 항공 경로등을 표시할때 마치 비행기가 이착륙하는듯한 곡선으로 도시들을 연결해준다. 이런 곡선을 보통 great circle (대권) 이라고 하는데, 여기서 R을 사용하여 이런 지도를 그리는 방법을 소개하려고한다.

우선 항공경로및 공항위치를 표시해줄 데이터가 필요한데, 아래 사이트에서 구할수있다.

데이터 출처 http://openflights.org/data.html

그리고, 대권과 지도를 구할때 필요한 팩키지는 아래와 같다.

library(maps)
library(plyr)
library(sp)

아래는 대권을 그리는데 필요한 데이터들을 정리하는 과정이다. 비행사, 항공노선 그리고 공항위치에 관련된 데이터가 따로 정리되어 있으므로 각 데이터 파일에서 필요한 필드를 뽑은후 하나의 데이터로 합쳐준다. 

airline <- read.csv("Data/Airlines.csv", sep = ",", header = TRUE)
airport <- read.csv("Data/Airport.csv", sep = ",", header = TRUE)
route <- read.csv("Data/Routes.csv", sep = ",", header = TRUE)

A <- airport[, c("ID", "Airport", "City", "Country", "Lat", "Lon")]
colnames(A) <- paste("Dep", colnames(A), sep = "_")
Dat <- merge(route[, c("Airline_ID", "Dep_Airport_ID", "Arr_Airport_ID")], A, 
    by.x = "Dep_Airport_ID", by.y = "Dep_ID")

A <- airport[, c("ID", "Airport", "City", "Country", "Lat", "Lon")]
colnames(A) <- paste("Arr", colnames(A), sep = "_")

Dat <- merge(Dat, A, by.x = "Arr_Airport_ID", by.y = "Arr_ID")

Dat <- merge(Dat, airline[, c("ID", "Airline", "Active")], by.x = "Airline_ID", 
    by.y = "ID")

데이터에는 전세계의 항공사, 항공노선, 공항위치등이 기록되어있으나, 여기서는 그 중 iHelp 멤버들이 거주하는 캐나다, 한국, 호주, 미국, 일본사이에 운영되는 항공노선만 표시해보겠다. 

DatX <- Dat[which(Dat$Dep_Country != Dat$Arr_Country), ]
DatX <- DatX[which(DatX$Dep_Country %in% c("Canada", "Australia", "South Korea", 
    "United States", "Japan") & DatX$Arr_Country %in% c("Canada", "Australia", 
    "South Korea", "United States", "Japan")), ]

항공경로의 색을 정할때 여러가지 방법이 있지만 여기서는 붉은색의 그라데이션 (gradation)을 사용하여 각기 다른 항공사들을 표시하겠다.

col_gradient <- colorRampPalette(c("white", "red"))
Colours <- col_gradient(length(unique(DatX$Airline_ID)))
DatX$Colour <- Colours[as.numeric(factor(as.character(DatX$Airline_ID)))]

아래에서는 대권을 구하는 방법인데, 간단하게 설명하자면 출발점과 도착점의 위도와 경도를 기준으로 그사이의 점들을 지정해주는 수만큼 구해서 연결해주면 대권이 된다. 여기서는 100개의 중간점들을 구하여 대권을 그렸다.

그리고 어떻한 세계지도를 사용하는냐에따라 경도의 지도상 한계점이 바뀌므로 여기에 맞춰서 계산을 다시해주어야 지도상 대권의 선이 끊어지지않고 잘 연결된다. 여기서는 태평양중심의 세계지도를 사용하므로 유럽과 아프리카의 서쪽이 분리되어 지도의 오른쪽으로 보여지게된다. 이것을 감안하여 아래처럼 경도를 다시 계산해주어야한다. 

GC_Dat <- list()

for (j in 1:nrow(DatX)) {
    X <- as.data.frame(gcIntermediate(DatX[j, c("Arr_Lon", "Arr_Lat")], DatX[j, 
        c("Dep_Lon", "Dep_Lat")], n = 100, addStartEnd = TRUE))

    R <- nrow(X[which(X$lon < 0), ])
    if (R == 0) {
        RR <- X
    }
    if (R > 0) {
        a <- ifelse(X$lon >= 0, X$lon, X$lon + 360)

        DF <- diff(a)
        L <- length(DF) - length(DF[which(DF < 0)])

        if (L == 0 | L == length(DF)) {
            X$lon <- a
            RR <- X
        }
        if (L != 0 & L != length(DF)) {
            aa <- X[which(X$lon >= 0), ]
            b <- X[which(X$lon < 0), ]
            if (nrow(b) > 0) {
                B <- b[nrow(b), ]

                b$lon <- b$lon + 360
                B$lon <- 360

                b <- rbind(b, B)
                b$id <- b$id + 0.5
            }

            RR <- rbind(aa, b)
        }
    }

    RR$Airline_ID <- DatX[j, "Airline_ID"]

    GC_Dat[[j]] <- merge(DatX[j, ], RR, by = "Airline_ID")
}

우선 태평양을 중심으로한 세계지도를 maps 팩키지를 사용하여 아래와같이 그릴수있다. 여기서는 바탕을 검은색으로 정하였는데 다른색을 사용할수도있다.

map("world2", col = "beige", bg = "black")

인공위성에서 찍은 사진같은 효과를 나타내기위해 인구가 상위 5%에 드는 도시들은 아래와같이 표시하였다.

data(world.cities)
Cities <- world.cities

Maj_Cities <- Cities[which(Cities$pop >= quantile(Cities$pop, prob = 0.95)), 
    ]

    360)

map("world2", col = "beige", bg = "black")
for (i in 1:length(GC_Dat)) {

}
