R (>=2.1.0)의 한국어 지원은 R Development Translation Teams 의 Korean 담당자 Chel Hee Lee (chl948@mail.usask.ca 또는 gnustats@gmail.com)에 진행되고 있습니다. 본 Language Support 페이지 페이지는 R (>=2.1.0)의 한국어 지원 프로세스와 방법, 그리고 진행사항들을 기술합니다 이에 대한 내용은 R 한국어 지원 메일링 리스트을 통해 확인하실 수 있으며, 해당 메일링의 가입은 ihelp-translators@lists.r-forge.r-project.org로 메일을 보낸다면 자동으로 이루어집니다. 또한, 모든 쓰레드는 ihelp-translators 알카이브를 통해 확인이 가능합니다

iHELP 참여자에게

iHELP 프로젝트는 아래와 같은 사항에 대해서 진행중입니다

  • R 메시지 한글 번역
  • R Documentation의 한국어판 제공
  • R 한국어 관련 버그리포팅
  • R 도움말 파일 제공
    • 본 프로젝트의 참여는 gnustats@gmail.com 으로 문의를 부탁드립니다.

    R 메시지의 한글 번역 참여자는 다음의 작업방법을 따라주셔야 하며, 개인이 스스로 모든 수정, 설치, 및 확인이 가능해야 합니다. 또한, 아래의 작업지침이 지켜지지 않거나 올바르게 번역되지 않은 메시지에 대해서는 삭제가 됨을 알려드립니다.

    이 작업방법은 2013년 04월에 배포될 R(≥3.0.0)에 적용되며, 그 이전 버전들인 R(≤ 2.15.3)에는 적용되지 않습니다. ihelp-translators 메일링 리스트를 참조바랍니다.

    현재 2013년 02월 18일 총 24개의 작업도메인이 있으나, 그 중 가장 기본인 base 패키지를 예로 듭니다.

    제일 먼저 소스트리를 iHELP 프로젝트로부터 다운받습니다. 

    	cd /src/library/base/po/ko.po

    이러한 작업을 위해서는 Fedora 그룹에서는 PoEdit를 추천합니다.

    파일을 열어보면 메시지 번역은 아래와 같은 형식을 가지고 있음을 알 수 있습니다. 

    	msgid "영어로 된 본래의 메시지가 있습니다"
	msgstr ""

    현재 2013년 2월 18일 R-3.0.0 을 위한 한국어 파일은 초기화 되어 있기 때문에 msgstr 부분은 아무런 메시지도 없이 "" 라고 되어 있습니다. msgid 에 있는 영어원문에 해당하는 올바른 한국어 번역을 아래와 같이 합니다.

    일반적인 설명을 위해 파일의 맨 처음 세가지 메시지를 이용하여 유형별로 설명합니다.

    유형 1. 

    	msgid "REPORT must be > 0 (method = \"BFGS\")"
	msgstr "method 에  \"BFGS\" 가 주어졌을때, REPORT 는 반드시 0 보다 큰 값이어야 합니다"

    유형 2. 

    	msgid "unknown 'type' in \"CG\" method of 'optim'"
	msgstr "'optim'에서 사용하는 method 옵션 중 \"CG\" 에서는 정해지지 않은 'type' 입니다 "

    위 두가지 유형은 R 내에서 최적화와 관련된 메시지로 optim() 이라는 함수와 관련된 것입니다. 그리고, 이 함수의 본래의 소스코드가 optim.c 입이며, 위의 메시지들은 C 코드로부터 나왔습니다. 따라서, optim.c 소스코드를 열어보니다. (아래는 소스코드입니다)

	typedef struct opt_struct
	{
			SEXP R_fcall;    /* function */
			SEXP R_gcall;    /* gradient */
			SEXP R_env;      /* where to evaluate the calls */
			double* ndeps;   /* tolerances for numerical derivatives */
			double fnscale;  /* scaling for objective */
			double* parscale;/* scaling for parameters */
			int usebounds;
			double* lower, *upper;
			SEXP names;	     /* names for par */
	} opt_struct, *OptStruct;

	static void genptry(int n, double *p, double *ptry, double scale, void *ex)
	{
			SEXP s, x;
			int i;
			OptStruct OS = (OptStruct) ex;
			PROTECT_INDEX ipx;

			if (!isNull(OS->R_gcall)) {
		/* user defined generation of candidate point */
		PROTECT(x = allocVector(REALSXP, n));
		for (i = 0; i < n; i++) {
				if (!R_FINITE(p[i]))
			error(_("non-finite value supplied by 'optim'"));
				REAL(x)[i] = p[i] * (OS->parscale[i]);
		}
		SETCADR(OS->R_gcall, x);
		PROTECT_WITH_INDEX(s = eval(OS->R_gcall, OS->R_env), &ipx);
		REPROTECT(s = coerceVector(s, REALSXP), ipx);
		if(LENGTH(s) != n)
				error(_("candidate point in 'optim' evaluated to length %d not %d"),
				LENGTH(s), n);
		for (i = 0; i < n; i++)
				ptry[i] = REAL(s)[i] / (OS->parscale[i]);
		UNPROTECT(2);
			}
			else {  /* default Gaussian Markov kernel */
		for (i = 0; i < n; i++)
				ptry[i] = p[i] + scale * norm_rand();  /* new candidate point */
			}
	}


	static double ** matrix(int nrh, int nch)
	{
			int   i;
			double **m;

			m = (double **) R_alloc((nrh + 1), sizeof(double *));
			for (i = 0; i <= nrh; i++)
		m[i] = (double*) R_alloc((nch + 1), sizeof(double));
			return m;
	}

	static double ** Lmatrix(int n)
	{
			int   i;
			double **m;

			m = (double **) R_alloc(n, sizeof(double *));
			for (i = 0; i < n; i++)
		m[i] = (double *) R_alloc((i + 1), sizeof(double));
			return m;
	}
